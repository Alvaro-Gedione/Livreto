    id: string;
    imageSource: string;
    width: number;
    height: number;
    ratio?: number;
}) => {

    image: string;
    width: number;
    height: number;

    rectWidth: number;
    rectHeight: number;
    framesWidth: number;
    framesHeight: number;
    spriteRealSize: {
        width: number;
        height: number;
    };
    textures: {
        attack3?: {
            offset: {
                x: number;
                y: number;
            };
            rectWidth: number;
            rectHeight: number;
            framesWidth: number;
            framesHeight: number;
            animations: ({ direction }: {
                direction: any;
            }) => any[];
        } | undefined;

                x: number;
                y: number;
            };

                direction: any;

                frameY: number;

            offset: {
                x: number;

            };

            framesHeight: number;

                direction: any;
            }) => any[];
        };

            offset: {
                x: number;
                y: number;
            };
            framesWidth: number;
            framesHeight: number;
            animations: ({ direction }: {
                direction: any;
            }) => any[];
        };
        skill: {
            framesWidth: number;
            framesHeight: number;
            animations: ({ direction }: {
                direction: any;
            }) => any[];
        };
        attack2: {
            offset: {
                x: number;
                y: number;
            };
            framesWidth: number;
            framesHeight: number;
            animations: ({ direction }: {
                direction: any;
            }) => any[];
        };
    };
};
